 45 wui.controls.control = ( function() {
 46     var count = 0;
 47     return function(type) {
 48         var idNumber = count;
 49         count += 1;
 50         var elementName = type ? type: "div";
 51         var mainElement = document.createElement(elementName);
 52         var show = function() {
 53             mainElement.style.display = 'block';
 54         };
 55         var hide = function() {
 56             mainElement.style.display = 'none';
 57         };
 58         var toggle = function() {
 59             if (mainElement.style.display === 'none') {
 60                 show();
 61             } else {
 62                 hide();
 63             }
 64         };
 65         
 66         var getControlNumber = function() {
 67             return idNumber;
 68         };
 69         
 70         var setOnClick = function(fn) {
 71             mainElement.onclick = fn;
 72         };
 73         
 74         var click = function(){
 75             var ev = document.createEvent('MouseEvents');
 76             ev.initEvent('click', true, true);
 77             mainElement.dispatchEvent(ev);
 78         };
 79         
 80         var css = (function(){
 81           var classes = [];
 82           
 83           var updateElement = function(){
 84             var i;
 85             var result = "";
 86             for(i=0; i < classes.length; i++){
 87               result += classes[i] + " ";
 88             }
 89             mainElement.className = result;
 90           };
 91           
 92           var toggleClass = function(className){
 93             var i;
 94             var exists = false;
 95             for(i=0; i < classes.length; i++){
 96               if(className === classes[i]){
 97                 exists = true;
 98                 break;
 99               }
100             }
101             if(!exists){
102               addClass(className);
103             } else {
104               removeClass(className)
105             }
106           };
107           
108           
109           var addClass = function(className){
110             var i;
111             var exists = false;
112             for(i=0; i < classes.length; i++){
113               if(className === classes[i]){
114                 exists = true;
115                 break;
116               }
117             }
118             if(!exists){
119               classes.push(className);
120               updateElement();
121             }
122           };
123           
124           var removeClass = function(className){
125             var i;
126             var newClasses = [];
127             var deleted = false;
128             for(i=0; i < classes.length; i++){
129               if(className !== classes[i]){
130                 newClasses.push(classes[i]);
131               }
132             }
133             classes = newClasses;
134             updateElement();
135           };
136           
137           var clear = function(className){
138             classes = [];
139             updateElement();
140           };
141           
142           return {
143             addClass: addClass,
144             removeClass: removeClass,
145             toggleClass: toggleClass,
146             clear: clear
147           }
148         }());
149         var setId = function(text) {
150             mainElement.id = text;
151         };
152         var getDomElement = function() {
153             return mainElement;
154         };
155         var appendControl = function(control) {
156             mainElement.appendChild(control.getDomElement());
157         };
158         return{
159             getDomElement: getDomElement,
160             getControlNumber: getControlNumber,
161             appendControl: appendControl,
162             show: show,
163             hide: hide,
164             toggle: toggle,
165             setOnClick: setOnClick,
166             click: click,
167             css: css,
168             setId: setId
169         };
170     };

239 wui.controls.panels.scroll = function() {
240     var that = wui.controls.control();
241     var id = "scroll__" + that.getControlNumber();
242     that.css.addClass("wui_position_flex");
243     that.setId(id);
244  
245     var scroller =  wui.controls.panels.standard();
246     that.appendControl(scroller);
247 	var mainDiv = that.getDomElement();
248     that.scroll = new iScroll(scroller.getDomElement(), {
249         desktopCompatibility: true
250     });
251 
252     that.setText = function(text) {
253         scroller.setText(text);
254         that.scroll.refresh();
255     };
256     that.appendControl= function(control) {
257         scroller.getDomElement().appendChild(control.getDomElement());
258         that.scroll.refresh();
259     };
260     that.show = function() {
261         that.getDomElement().style.display = 'block';
262         that.scroll.refresh();
263     };
264     that.scroll.refresh();
265 
266 	// Prevent other parts of the page from being draggable
267     document.addEventListener('touchmove', function(e) {
268         e.preventDefault();
269     },
270     false);
271 
272     that.clear = function() {
273         scroller.getDomElement().innerHTML = "";
274     };
275     that.getDomElement = function() {
276         return mainDiv;
277     };
278     return that;

413  wui.controls.menu = {}; /*
414  * Create a menuItem to load in a menuBar
415  * Dependencies: 
416  *  wui.js
417  *  wui_controls.js
418  *  wui_controls_menu.js
419  * 
420  *  wui_controls_menu_menuItem.css
421  */
422 wui.controls.menu.menuItem = function() {
423     var that = wui.controls.control("span");
424     that.css.addClass("wui_position_flex");
425     that.css.addClass("wui_control_menuItem");
426     var icon = wui.controls.control();
427     icon.css.addClass("wui_control_menuItem_icon");
428     var selected = false;
429     var panelControl;
430     var normalIcon = "";
431     var selectedIcon = "";
432     var a = wui.controls.control("a");
433     that.setText = function(text) {
434         a.getDomElement().innerHTML = text;
435     };
436     that.setPanel = function(panel) {
437         panelControl = panel;
438     };
439     that.getPanel = function(){
440     	return panelControl;
441     };
442     that.showPanel = function() {
443         if (panelControl) {
444             panelControl.show();
445         }
446     };
447     that.hidePanel = function() {
448         if (panelControl) {
449             panelControl.hide();
450         }
451     };
452     that.setIconControl = function(text) {
453         normalIcon = text;
454         if (!selected) {
455             icon.getDomElement().innerHTML = normalIcon;
456         }
457     };
458     that.setSelectedIconControl = function(text) {
459         selectedIcon = text;
460     };
461     that.select = function() {
462         if (selectedIcon.length > 1) {
463             icon.getDomElement().innerHTML = selectedIcon;
464         }
465         selected = true;
466         that.css.addClass("wui_control_menuItem_active");
467         that.showPanel();
468     };
469     that.deselect = function() {
470         icon.getDomElement().innerHTML = normalIcon;
471         selected = false;
472         that.css.removeClass("wui_control_menuItem_active");
473         that.hidePanel();
474     };
475     that.show = function() {
476         that.getDomElement().style.display = '-webkit-box';
477     };
478     that.appendControl(icon);
479     that.appendControl(a);
480     return that;
481 };
482  /*
483  * Create a menubar
484  * Dependencies: 
485  *  wui.js
486  *  wui_controls.js
487  *  wui_controls_menu.js
488  * 
489  *  wui_controls_menu_menuBar.css
490  */
491 wui.controls.menu.menuBar = function() {
492     var that = wui.controls.control("nav");
493     var items = [];
494     var i;
495     that.css.addClass("wui_position_hbox");
496     that.css.addClass("wui_control_menuBar");
497     var activeItem;
498 
499     that.setActive = function(menuItem) {
500         for (i = 0; i < items.length; i++) {
501             items[i].deselect();
502         }
503         menuItem.select();
504         activeItem = menuItem;
505     };
506     
507     that.getActive = function(){
508     	return activeItem;
509     };
510     
511     that.items = (function() {
512         var add = function(menuItem) {
513             menuItem.setOnClick( function() {
514                 that.setActive(menuItem);
515             });
516             items.push(menuItem);
517             that.appendControl(menuItem);
518         };
519         return{
520             add: add
521         };
522     }());
523     
524     return that;
